static int buscaBinaria(const vector<int>& vetor, int elemento) {
    int inicio = 0;
    int fim = vetor.size() - 1;

    while (inicio <= fim) {
        int meio = inicio + (fim - inicio) / 2;

        if (vetor[meio] == elemento)
            return meio;

        if (vetor[meio] < elemento)
            inicio = meio + 1;
        else
            fim = meio - 1;
    }

    return -1;
}
